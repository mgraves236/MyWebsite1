void NQueens(int n, int column, int board[]) {
    int row, i;
    if (column == n) {
        printf("{");
        for (i = 0; i < n; i++) {
            printf(" %d ", board[i]);
        }
        printf("}\n");
        printf("\n");
     }

    for (row = 0; row < n; row++) {
         i = 0;
         while (i < column) {
            if (row==board[i] || abs(board[i]-row)==column-i) break;
            i++;
          }
          if (i== column) {
              board[column]=row;
              NQueens(n, column+1, board);
          }
    }
}
